// Donnes stockes en mmoire

let rapaces = [];

let entries = [];

 

// Initialisation

document.addEventListener('DOMContentLoaded', function() {

loadData();

updateRapaceSelects();

updateEntryList();

document.getElementById('event-time').value = new Date().toLocaleTimeString('fr-FR', {hour: '2-digit', minute: '2-digit'});

// Gestion des onglets

function showTab(tabName) {

document.querySelectorAll('.tab').forEach(tab => tab.classList.remove('active'));

document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));

event.target.classList.add('active');

document.getElementById(tabName).classList.add('active');

if (tabName === 'stats') {

updateStats();

} else if (tabName === 'rapaces') {

updateRapaceList();

}

}

 

// Actions rapides

function quickAction(type) {

const rapaceSelect = document.getElementById('rapace-select');

if (!rapaceSelect.value) {

alert('Veuillez d\'abord slectionner un rapace');

return;

}

 

const eventType = document.getElementById('event-type');

const eventValue = document.getElementById('event-value');

switch(type) {

case 'pesee':

eventType.value = 'poids';

eventValue.placeholder = 'Ex: 1.2kg';

break;

case 'repas':

eventType.value = 'nourriture';

eventValue.placeholder = 'Ex: 200g de caille';

break;

case 'vol':

eventType.value = 'vol';

eventValue.placeholder = 'Ex: 1h30';

break;

case 'sante':

eventType.value = 'sante';

eventValue.placeholder = 'Ex: Contrle serres';

break;

}

eventValue.focus();

}

 

// Prvisualisation d'image

function previewImage(input) {

const preview = document.getElementById('photo-preview');

if (input.files && input.files[0]) {

const reader = new FileReader();

reader.onload = function(e) {

preview.innerHTML = `<img src="${e.target.result}" alt="Aperu">`;

};

reader.readAsDataURL(input.files[0]);

} else {

preview.innerHTML = '';

}

}

 

// Ajouter un rapace

function addRapace() {

const name = document.getElementById('rapace-name').value;

const type = document.getElementById('rapace-type').value;

const photoInput = document.getElementById('rapace-photo');

if (!name) {

alert('Veuillez entrer un nom');

return;

}

const rapace = {

id: Date.now(),

name: name,

type: type,

created: new Date().toISOString(),

photo: null

};

// Traitement de l'image si prsente

if (photoInput.files && photoInput.files[0]) {

const reader = new FileReader();

reader.onload = function(e) {

rapace.photo = e.target.result;

rapaces.push(rapace);

saveData();

updateRapaceSelects();

updateRapaceList();

clearAddRapaceForm();

};

reader.readAsDataURL(photoInput.files[0]);

} else {

rapaces.push(rapace);

saveData();

updateRapaceSelects();

updateRapaceList();

clearAddRapaceForm();

}

}

 

// Vider le formulaire d'ajout

function clearAddRapaceForm() {

document.getElementById('rapace-name').value = '';

document.getElementById('rapace-type').value = 'faucon';

document.getElementById('rapace-photo').value = '';

document.getElementById('photo-preview').innerHTML = '';

}

 

// Supprimer un rapace

function deleteRapace(id) {

if (confirm('tes-vous sr de vouloir supprimer ce rapace ?')) {

rapaces = rapaces.filter(r => r.id !== id);

entries = entries.filter(e => e.rapaceId !== id);

saveData();

updateRapaceSelects();

updateRapaceList();

}

}

 

// Ajouter une entre

function addEntry() {

const rapaceId = document.getElementById('rapace-select').value;

const type = document.getElementById('event-type').value;

const value = document.getElementById('event-value').value;

const time = document.getElementById('event-time').value;

const notes = document.getElementById('event-notes').value;

if (!rapaceId || !value) {

alert('Veuillez slectionner un rapace et entrer une valeur');

return;

}

const entry = {

id: Date.now(),

rapaceId: parseInt(rapaceId),

type: type,

value: value,

time: time,

notes: notes,

date: new Date().toISOString()

};

entries.push(entry);

saveData();

updateEntryList();

// Rinitialiser le formulaire

document.getElementById('event-value').value = '';

document.getElementById('event-notes').value = '';

document.getElementById('event-time').value = new Date().toLocaleTimeString('fr-FR', {hour: '2-digit', minute: '2-digit'});

}

 

// Mettre jour la liste des rapaces dans les slecteurs

function updateRapaceSelects() {

const selects = ['rapace-select', 'stats-rapace-select'];

selects.forEach(selectId => {

const select = document.getElementById(selectId);

const currentValue = select.value;

select.innerHTML = '<option value="">Slectionner un rapace</option>';

rapaces.forEach(rapace => {

const option = document.createElement('option');

option.value = rapace.id;

option.textContent = `${rapace.name} (${rapace.type})`;

select.appendChild(option);

});

select.value = currentValue;

});

}

 

// Mettre jour la liste des entres

function updateEntryList() {

const selectedRapaceId = document.getElementById('rapace-select').value;

const entryList = document.getElementById('entry-list');

if (!selectedRapaceId) {

entryList.innerHTML = '<p style="text-align: center; color: #7f8c8d;">Slectionnez un rapace pour voir ses entres</p>';

return;

}

const rapaceEntries = entries

.filter(entry => entry.rapaceId == selectedRapaceId)

.sort((a, b) => new Date(b.date) - new Date(a.date))

.slice(0, 10);

entryList.innerHTML = '';

rapaceEntries.forEach(entry => {

const div = document.createElement('div');

div.className = 'entry-item';

const date = new Date(entry.date);

const formattedDate = date.toLocaleDateString('fr-FR') + ' ' + entry.time;

div.innerHTML = `

<div class="date">${formattedDate}</div>

<div class="content">

<strong>${getEventTypeLabel(entry.type)}</strong>: ${entry.value}

${entry.notes ? `<br><em>${entry.notes}</em>` : ''}

</div>

`;

entryList.appendChild(div);

});

if (rapaceEntries.length === 0) {

entryList.innerHTML = '<p style="text-align: center; color: #7f8c8d;">Aucune entre pour ce rapace</p>';

}

}

 

// Mettre jour les statistiques

function updateStats() {

const selectedRapaceId = document.getElementById('stats-rapace-select').value;

if (!selectedRapaceId) {

document.getElementById('last-weight').textContent = '--';

document.getElementById('meals-today').textContent = '--';

document.getElementById('flight-time').textContent = '--';

document.getElementById('total-entries').textContent = '--';

return;

}

const rapaceEntries = entries.filter(entry => entry.rapaceId == selectedRapaceId);

const today = new Date().toDateString();

const todayEntries = rapaceEntries.filter(entry => new Date(entry.date).toDateString() === today);

// Dernier poids

const weightEntries = rapaceEntries.filter(entry => entry.type === 'poids').sort((a, b) => new Date(b.date) - new Date(a.date));

document.getElementById('last-weight').textContent = weightEntries.length > 0 ? weightEntries[0].value : '--';

// Repas aujourd'hui

const mealsToday = todayEntries.filter(entry => entry.type === 'nourriture').length;

document.getElementById('meals-today').textContent = mealsToday;

// Temps de vol aujourd'hui

const flightEntries = todayEntries.filter(entry => entry.type === 'vol');

document.getElementById('flight-time').textContent = flightEntries.length > 0 ? flightEntries[0].value : '--';

// Total entres

document.getElementById('total-entries').textContent = rapaceEntries.length;

}

 

// Mettre jour la liste des rapaces

function updateRapaceList() {

const rapaceList = document.getElementById('rapace-list');

rapaceList.innerHTML = '';

rapaces.forEach(rapace => {

const div = document.createElement('div');

div.className = 'rapace-item';

const entriesCount = entries.filter(entry => entry.rapaceId === rapace.id).length;

// Cration de l'lment photo

let photoElement = '';

if (rapace.photo) {

photoElement = `<img src="${rapace.photo}" alt="${rapace.name}" class="rapace-photo">`;

} else {

// Utilise la premire lettre du nom comme placeholder

const firstLetter = rapace.name.charAt(0).toUpperCase();

photoElement = `<div class="rapace-photo-placeholder">${firstLetter}</div>`;

}

div.innerHTML = `

${photoElement}

<div class="rapace-info">

<h3>${rapace.name}</h3>

<p>${rapace.type} ${entriesCount} entres</p>

</div>

<button class="delete-btn" onclick="deleteRapace(${rapace.id})">Supprimer</button>

`;

rapaceList.appendChild(div);

});

if (rapaces.length === 0) {

rapaceList.innerHTML = '<p style="text-align: center; color: #7f8c8d;">Aucun rapace ajout</p>';

}

}

 

// Libells des types d'vnements

function getEventTypeLabel(type) {

const labels = {

'poids': 'Poids',

'nourriture': 'Nourriture',

'vol': 'Vol',

'sante': 'Sant',

'comportement': 'Comportement'

};

return labels[type] || type;

}

}

}
